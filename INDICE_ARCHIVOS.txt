ASIGNAR MÓDULO A USUARIO:
  INSERT INTO USUARIO_MODULO (IDUSUARIO_MODULO, IDUSUARIO, IDMODULO_id, 
  PERMISOS, ASIGNADO_POR, ACTIVO)
  VALUES (NEWID(), 'juan', 'MOD002', '["read","write"]', 'admin', 1);

VER MÓDULOS DE USUARIO:
  SELECT um.IDUSUARIO, m.NOMBRE, um.PERMISOS
  FROM USUARIO_MODULO um
  JOIN MODULO m ON um.IDMODULO_id = m.IDMODULO
  WHERE um.IDUSUARIO = 'juan' AND um.ACTIVO = 1;

DESASIGNAR MÓDULO:
  UPDATE USUARIO_MODULO
  SET ACTIVO = 0
  WHERE IDUSUARIO = 'juan' AND IDMODULO_id = 'MOD002';

CAMBIAR PERMISOS:
  UPDATE USUARIO_MODULO
  SET PERMISOS = '["read","write","delete"]'
  WHERE IDUSUARIO = 'juan' AND IDMODULO_id = 'MOD002';
